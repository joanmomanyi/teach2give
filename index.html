Question 1: FizzBuzz

Write a program that prints the numbers from 1 to 100.

  • For multiples of 3, print "Fizz";
  • For multiples of 5, print "Buzz";
  • And for numbers that are multiples of both 3 and 5, print "FizzBuzz".

Solution

//Loop through all numbers from 1 to 100 //Current number is assigned variable i for (let i = 1; i <= 100; i++) { //We use modulus division (remainder) //If the remainder === 0 means number //is divisible //If i is divisible by 3 & 5; print "FizzBuzz" if (i % 3 === 0 && i % 5 === 0) { console.log("FizzBuzz"); //Else if i is divisible by 3, print "Fizz" } else if (i % 3 === 0) { console.log("Fizz"); //Else if i is divisible by 5, print "Buzz" } else if (i % 5 === 0) { console.log("Buzz"); //Otherwise, print the number i } else { console.log(i); } }

Question 2

Write a program to generate the Fibonacci sequence up to 100

The Fibonacci sequence is a series of numbers in which each number is the sum of the two preceding ones.
For example: 0, 1, 1, 2, 3, 5, 8, 13...

Solution

// Initialize the Fibonacci sequence with the first number, 0. const fibonacciSequence = [0]; // Initialize the first two Fibonacci numbers (0 and 1) let num1 = 0; let num2 = 1; // Start an infinite loop to generate the Fibonacci sequence. while (true) { // Calculate the next Fibonacci number by adding the previous two. const nextNum = num1 + num2; // Check if the next number exceeds 100. if (nextNum > 100) { // If it does, exit the loop. break; } // Add the calculated number to the Fibonacci sequence array. fibonacciSequence.push(nextNum); // Update num1 and num2 for the next iteration. num1 = num2; num2 = nextNum; } // Output the generated Fibonacci sequence to the console. console.log(fibonacciSequence);

Question 3: Power of Two

Write a program that takes an integer as input and returns true if the input is a power of two.

Examples:

8=> returns true
6=> returns false

Solution

/** * Checks if a given integer is a power of two. * * @param {number} n The integer to check. * @returns {boolean} true if the integer is a power of 2 o false if not. */ function isPowerOfTwo(n) { // Handle edge cases: // 1. Numbers less than or equal to 0 are not powers of two. if (n <= 0) { return false; } // 2. 1 is a power of 2 (2^0). if (n === 1) { return true; } // Repeatedly divide n by 2 until it becomes 1 or an odd number. // Using modulus division ensures the remainder is 0 otherwise the number is odd while (n % 2 === 0) { n = n/2; } // If n is 1 after repeated division, it was a power of two. if (n === 1) { return true; } else{ return false; } } // Example usage: console.log(isPowerOfTwo(8)); // Output: true console.log(isPowerOfTwo(6)); // Output: false console.log(isPowerOfTwo(1)); //output: true console.log(isPowerOfTwo(0)); //output: false console.log(isPowerOfTwo(-4)); //output: false
